testing <- read.csv(url(testUrl), na.strings=c("NA","#DIV/0!",""))

Preprocessing begins next

training_nas <- training[,(colSums(is.na(training)) == 0)]
testing_nas <- testing[,(colSums(is.na(testing)) == 0)]

Remove zero variance.

zvar <- nearZeroVar(training_nas,saveMetrics=TRUE)
trainingNZ<- training_nas[,!zvar$nzv]
testingNZ <- testing_nas[,!zvar$nzv]

#metaData_train <- c("user_name","raw_timestamp_part_1","raw_timestamp_part_2","cvtd_timestamp","num_window","problem_id")

#Cross-validation The next section partitions the training data into a training subset and a validation subset which is named Testing_A, not to be confused with the downloaded testing dataset of 20 observations used to generate the test files at the end. The Random Forest method has its own internal cross-validation algorithms which protect against overfitting.

inTrain <- createDataPartition(y=training$classe, p=0.6, list=FALSE)
training_A <- training_clean[inTrain, ]
testing_A <- training_clean[-inTrain, ]
dim(training_A)

## [1] 11776    53

dim(testing_A)

## [1] 7846   53

#Model Building The next section builds the random forest model and then uses it to predict the “classe” variable in the validation partition. It is then used to generate a confusion matrix which will provide measures of uncertainty

RFmodFit <- randomForest(classe ~. , data=training_A, importance=TRUE)

#Predicting:

predictionsRF1 <- predict(RFmodFit, testing_A, type = "class")

#The acid test:  Generating a confusion matrix
confusionMatrix(predictionsRF1, testing_A$classe)

pml_write_files = function(x){
  n = length(x)
  for(i in 1:n){
    filename = paste0("problem_id_",i,".txt")
    write.table(x[i],file=filename,quote=FALSE,row.names=FALSE,col.names=FALSE)
  }
}
